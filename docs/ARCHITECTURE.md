// App.jsx State
{
  osList: [],              // Available OS targets
  packages: [],            // Available packages
  selectedPackages: [],    // User selection
  imageName: '',          // Target image name
  osTarget: '',           // Selected OS
  buildId: null,          // Current build ID
  status: null,           // Build status
  isBuilding: false,      // Build in progress
  commandPreview: '',     // Preview of docker command
  buildResult: null       // Success data: {imageName, osTarget, packages}

  logs: [],               // Array of log lines
  autoScroll: true        // Auto-scroll enabled
}

builds: Map<id, {
  logs: [],               // Captured output
  status: string,         // 'running' | 'completed' | 'failed'
  error: string | null,   // Error message
  emitter: EventEmitter,  // For streaming
  command: string,        // The docker build command
  imageName: string,      // Target image
  osTarget: string,       // OS target
  packages: string[]      // Included packages
}>

## ğŸ“¡ API Endpoints

| Method | Endpoint | Purpose | Response |
|--------|----------|---------|----------|
| GET | `/api/health` | Health check | `{ status, time }` |
| GET | `/api/os` | List OS targets | `["azlinux3", ...]` |
| GET | `/api/packages` | List packages | `["curl", "bash", ...]` |
| POST | `/api/build` | Start build | `{ buildId, command }` |
